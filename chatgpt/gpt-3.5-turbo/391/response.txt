```
import org.ietf.jgss.*;
import java.io.*;

public class GSSContextExample {
    public void wrapData(InputStream inStream, OutputStream outStream, GSSContext context) throws GSSException, IOException {
        MessageProp msgProp = new MessageProp(0, true);
        byte[] inBytes = inStream.readAllBytes();
        byte[] wrappedBytes = context.wrap(inBytes, 0, inBytes.length, msgProp);
        outStream.write(wrappedBytes);
    }
}
```
