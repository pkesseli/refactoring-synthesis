import java.awt.List;

public class Example {
   public static void main(String[] args) {
      List myList = new List();

   }
}
```

import java.awt.List;

public class Example {
   public static void main(String[] args) {
      List myList = new List();

      myList.add("item3");

      myList.removeAll(); // recommended method
   }
}
```
