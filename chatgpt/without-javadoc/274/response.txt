import javax.management.MBeanServer;
import javax.management.ObjectName;
import javax.management.Query;

    private MBeanServer mBeanServer;

    public AttributeValueExp getAttributeValueExp(ObjectName objectName, String attributeName) {
        AttributeValueExp attributeValueExp = new AttributeValueExp();
        attributeValueExp.setTargetObject(objectName);
        attributeValueExp.setAttributeName(attributeName);
        attributeValueExp.setMBeanServer(mBeanServer);

        Query query = Query.eq(Query.attr(attributeName), Query.value(attributeValueExp));
