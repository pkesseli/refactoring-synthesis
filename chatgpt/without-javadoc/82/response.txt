```java
import java.awt.Frame;
import java.awt.List;

public class ExampleFrame extends Frame {
```

After refactoring:

```java
import java.awt.Dimension;
import java.awt.Frame;
import java.awt.List;

public class ExampleFrame extends Frame {

    private List list;

    public ExampleFrame() {
        list = new List();
        add(list);
        pack();  // Use pack() instead of setting size manually
        setVisible(true);
    }

    @Override
    public Dimension getPreferredSize() {
        return new Dimension(200, 200);  // Set preferred size here instead
    }
}
```
